// app/api/stripe/webhook/route.ts
import Stripe from "stripe";
import { headers } from "next/headers";
import { NextResponse } from "next/server";

export const runtime = "nodejs"; // required for raw body

export async function POST(req: Request) {
  const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
    apiVersion: "2023-10-16",
  });

  const body = await req.text();
  const sig = headers().get("stripe-signature")!;

  try {
    const event = stripe.webhooks.constructEvent(
      body,
      sig,
      process.env.STRIPE_WEBHOOK_SECRET!
    );

    // üî• Handle successful payments
    if (event.type === "checkout.session.completed") {
      const session = event.data.object;

      console.log("üí∞ Payment complete:", session);

      // TODO: Mark the user as paid, unlock audio download, etc.
      // you can use session.metadata.audioUrl
    }

    return NextResponse.json({ received: true });
  } catch (err: any) {
    console.error("‚ùå Webhook error:", err.message);
    return new NextResponse(`Webhook Error: ${err.message}`, { status: 400 });
  }
}
